### 步骤 1: 视频画面点击选择

在 QML 的 `CameraView` 组件中添加 MouseArea：

`gui/frontend/components/CameraView.qml`：

```qml
import QtQuick
import QtQuick.Controls

Item {
    id: root
    property alias source: image.source

    signal bottlecapClicked(int capId)

    Image {
        id: image
        anchors.fill: parent
        fillMode: Image.PreserveAspectFit
        cache: false

        MouseArea {
            anchors.fill: parent
            onClicked: function(mouse) {
                // 从 bridge 获取检测框列表, 匹配点击坐标
                var detections = bridge.currentDetections
                var imgW = image.paintedWidth
                var imgH = image.paintedHeight
                var imgX = (root.width - imgW) / 2
                var imgY = (root.height - imgH) / 2

                for (var i = 0; i < detections.length; i++) {
                    var d = detections[i]
                    if (mouse.x >= imgX + d.x1 * imgW / 640 &&
                        mouse.x <= imgX + d.x2 * imgW / 640 &&
                        mouse.y >= imgY + d.y1 * imgH / 480 &&
                        mouse.y <= imgY + d.y2 * imgH / 480) {
                        root.bottlecapClicked(d.id)
                        return
                    }
                }
            }
        }
    }
}
```

### 步骤 3: 坐标变换标定工具

创建 `extension/robot/calibrator.py`：

```python
"""半自动标定工具: 相机坐标系 → 机械臂基座坐标系"""
import json
import numpy as np
from pathlib import Path

from core.config.settings import get, load_config


class HandEyeCalibrator:
    """手眼标定: 求解相机在机械臂基座系中的位置偏移"""

    def __init__(self, save_path=None):
        self._points_cam = []
        self._points_robot = []
        self._save_path = save_path or Path(__file__).parent.parent.parent / "settings" / "calibration.json"
        self._loaded = False
        self._offset_x = get("coordinator.cam_offset_x", 0)
        self._offset_y = get("coordinator.cam_offset_y", 200)
        self._offset_z = get("coordinator.cam_offset_z", 150)

    def add_point(self, cam_x, cam_y, cam_z, robot_x, robot_y, robot_z):
        """添加一组标定点对"""
        self._points_cam.append([cam_x, cam_y, cam_z])
        self._points_robot.append([robot_x, robot_y, robot_z])
        print(f"[Calibrator] 第{len(self._points_cam)}组点对: "
              f"相机({cam_x:.0f},{cam_y:.0f},{cam_z:.0f}) → "
              f"机械臂({robot_x:.0f},{robot_y:.0f},{robot_z:.0f})")

    def solve(self):
        """最少二乘法求解偏移量"""
        if len(self._points_cam) < 3:
            print("[Calibrator] 需要至少3组标定点")
            return False

        cam = np.array(self._points_cam)
        robot = np.array(self._points_robot)

        # 简化: 计算平均偏移量
        offsets = robot - cam
        self._offset_x = float(np.mean(offsets[:, 0]))
        self._offset_y = float(np.mean(offsets[:, 1]))
        self._offset_z = float(np.mean(offsets[:, 2]))

        print(f"[Calibrator] 标定完成:")
        print(f"  offset_x = {self._offset_x:.1f} mm")
        print(f"  offset_y = {self._offset_y:.1f} mm")
        print(f"  offset_z = {self._offset_z:.1f} mm")

        self._save()
        return True

    def _save(self):
        data = {
            "cam_offset_x": self._offset_x,
            "cam_offset_y": self._offset_y,
            "cam_offset_z": self._offset_z,
            "num_points": len(self._points_cam),
        }
        self._save_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self._save_path, "w", encoding="utf-8") as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
        print(f"[Calibrator] 标定结果已保存到 {self._save_path}")

    def load(self):
        if self._save_path.exists():
            with open(self._save_path, "r", encoding="utf-8") as f:
                data = json.load(f)
                self._offset_x = data.get("cam_offset_x", self._offset_x)
                self._offset_y = data.get("cam_offset_y", self._offset_y)
                self._offset_z = data.get("cam_offset_z", self._offset_z)


# 标定流程示例
if __name__ == "__main__":
    calib = HandEyeCalibrator()

    print("""
    手动标定流程:
    1. 放置一个标记物在相机视野中
    2. 运行检测获取相机的 (cam_x, cam_y, cam_z)
    3. 手动测量标记物在机械臂基座系中的坐标 (robot_x, robot_y, robot_z)
    4. 调用 calib.add_point(cam_x, cam_y, cam_z, robot_x, robot_y, robot_z)
    5. 重复步骤 1-4 至少 3 次
    6. 调用 calib.solve() 计算偏移量
    """)
```

```python
"""端到端视觉引导抓取测试"""
import sys
sys.path.insert(0, '.')

from extension.vision.camera import create_camera
from extension.vision.detector import BottlecapDetector
from extension.robot.serial_client import create_arm_client
from extension.robot.coordinator import CoordinateTransformer
from tools.vision_tools import VisionToolContext, register as reg_vis
from tools.robot_tools import RobotToolContext, register as reg_rob
from tools import call_tool

reg_vis()
reg_rob()

def test_full_flow():
    camera = create_camera(use_mock=True)
    camera.start()

    detector = BottlecapDetector()
    arm = create_arm_client(use_mock=True)
    arm.connect()
    coordinator = CoordinateTransformer()

    RobotToolContext.init(arm, coordinator)

    import time
    time.sleep(1)

    color, depth, intrinsics = camera.get_frames()
    detections = detector.detect(color, depth, intrinsics)

    for d in detections:
        d["color"] = detector.extract_color(color, d)

    VisionToolContext.update(detections)
    print(f"\n检测到 {len(detections)} 个瓶盖:")

    for d in detections:
        info = d.copy()
        info.pop("bbox", None)
        print(f"  #{d['id']}: {info}")

    if detections:
        target_id = max(detections, key=lambda d: d["confidence"])["id"]
        print(f"\n选择置信度最高的瓶盖 #{target_id}")

        result = call_tool("select_bottlecap", {"id": target_id})
        print(f"  select: {result}")

        info = call_tool("get_bottlecap_info", {"id": target_id})
        print(f"  info: {info}")

        target = VisionToolContext.get_detections()
        target = next((d for d in target if d["id"] == target_id), None)
        if target and "X" in target:
            result = call_tool("move_arm_to", {
                "x": target["X"], "y": target["Y"], "z": target["Z"]
            })
            print(f"  move: {result}")

        result = call_tool("grasp", {})
        print(f"  grasp: {result}")

        result = call_tool("get_arm_status", {})
        print(f"  status: {result}")

    camera.stop()
    arm.disconnect()
    print("\n端到端测试完成!")

if __name__ == "__main__":
    test_full_flow()
```

## 常见问题

**Q: 点击瓶盖后坐标偏差大？**
A: 可能是标定不准确。重新运行 `calibrator.py` 标定流程，至少采集 4 组点对。

**Q: move_arm_to 返回 "超出工作空间"？**
A: 检查 `coordinator.is_workspace_valid()` 的判断条件（默认 50-280mm），根据实际机械臂参数调整。

**Q: 坐标变换后机械臂运动反向？**
A: 检查 `coordinator.camera_to_robot()` 中坐标轴的对应关系，可能某轴需要取反。
