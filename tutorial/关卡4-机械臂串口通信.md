### 步骤 1: 定义协议常量

`extension/robot/protocol.py`：

```python
"""机械臂串口协议定义"""
FRAME_HEADER = 0xAA
FRAME_LENGTH = 0x07

CMD_GRASP = 0    # 抓取
CMD_RELEASE = 1  # 释放

TARGET_MIN = -127
TARGET_MAX = 127

def checksum(data: bytes) -> int:
    """计算校验和: 前6字节累加取低8位"""
    return sum(data) & 0xFF

def build_frame(cmd: int, x: int, y: int, z: int) -> bytes:
    """构建7字节协议帧"""
    x = max(TARGET_MIN, min(TARGET_MAX, x))
    y = max(TARGET_MIN, min(TARGET_MAX, y))
    z = max(TARGET_MIN, min(TARGET_MAX, z))

    payload = bytearray([
        FRAME_HEADER,
        FRAME_LENGTH,
        cmd & 0xFF,
        x & 0xFF,
        y & 0xFF,
        z & 0xFF,
    ])
    payload.append(checksum(payload))
    return bytes(payload)
```

### 步骤 2: 实现串口客户端

`extension/robot/serial_client.py`：

```python
"""机械臂串口通信客户端"""
import time
import threading
import serial
import serial.tools.list_ports

from extension.robot.protocol import build_frame, CMD_GRASP, CMD_RELEASE


class ArmError(Exception):
    """机械臂通信异常"""


class RoboticArmClient:
    """封装 7 字节协议, 管理串口连接"""

    def __init__(self, port="COM3", baudrate=115200, timeout=0.5):
        self._port = port
        self._baudrate = baudrate
        self._timeout = timeout
        self._ser = None
        self._lock = threading.Lock()
        self._connected = False

        self._status = {
            "position": {"a": 0, "b": 0, "c": 0, "g": 0},
            "gripper": "unknown",
            "busy": False,
            "has_object": False,
        }

    @property
    def connected(self):
        return self._connected

    @property
    def status(self):
        return self._status.copy()

    @staticmethod
    def list_ports():
        """列出可用串口"""
        ports = serial.tools.list_ports.comports()
        return [(p.device, p.description) for p in ports]

    def connect(self):
        if self._connected:
            return
        try:
            self._ser = serial.Serial(
                port=self._port,
                baudrate=self._baudrate,
                bytesize=serial.EIGHTBITS,
                parity=serial.PARITY_NONE,
                stopbits=serial.STOPBITS_ONE,
                timeout=self._timeout,
            )
            self._connected = True
            print(f"[Arm] 串口已连接: {self._port} @ {self._baudrate}")
        except serial.SerialException as e:
            raise ArmError(f"串口连接失败: {e}")

    def disconnect(self):
        if self._ser and self._ser.is_open:
            self._ser.close()
        self._connected = False
        print("[Arm] 串口已断开")

    def send_grasp(self, x: int, y: int, z: int):
        """发送抓取指令: 移动到(x,y,z) + 抓取"""
        frame = build_frame(CMD_GRASP, x, y, z)
        self._send_frame(frame)
        print(f"[Arm] 抓取指令: x={x}, y={y}, z={z}")

    def send_release(self, x: int, y: int, z: int):
        """发送释放指令: 移动到(x,y,z) + 释放"""
        frame = build_frame(CMD_RELEASE, x, y, z)
        self._send_frame(frame)
        print(f"[Arm] 释放指令: x={x}, y={y}, z={z}")

    def emergency_stop(self):
        """紧急停止: 断开串口"""
        self.disconnect()
        print("[Arm] 紧急停止!")

    def _send_frame(self, frame: bytes):
        with self._lock:
            if not self._connected:
                raise ArmError("串口未连接")
            try:
                self._ser.write(frame)
                self._ser.flush()
            except serial.SerialException as e:
                self._connected = False
                raise ArmError(f"串口写入失败: {e}")

    def update_status(self, angles=None):
        """更新机械臂状态(供回调使用)"""
        if angles:
            self._status["position"].update(angles)

class MockArmClient(RoboticArmClient):
    """离线测试用模拟机械臂"""

    def __init__(self):
        super().__init__()
        self._connected = True
        self._mock_x = 0
        self._mock_y = 0
        self._mock_z = 0

    def connect(self):
        self._connected = True
        print("[MockArm] 模拟机械臂已连接")

    def disconnect(self):
        self._connected = False
        print("[MockArm] 模拟机械臂已断开")

    def send_grasp(self, x, y, z):
        self._mock_x, self._mock_y, self._mock_z = x, y, z
        self._status["gripper"] = "closed"
        self._status["has_object"] = True
        print(f"[MockArm] 模拟抓取: x={x}, y={y}, z={z}")

    def send_release(self, x, y, z):
        self._mock_x, self._mock_y, self._mock_z = x, y, z
        self._status["gripper"] = "open"
        self._status["has_object"] = False
        print(f"[MockArm] 模拟释放: x={x}, y={y}, z={z}")

    @property
    def connected(self):
        return True


def create_arm_client(use_mock=False, port="COM3"):
    if use_mock:
        return MockArmClient()
    return RoboticArmClient(port=port)

"""相机坐标系 → 机械臂基座坐标系 变换"""
import math


class CoordinateTransformer:
    """坐标系变换 + 协议量化"""

    def __init__(self, offset_x=0, offset_y=200, offset_z=150):
        # 相机光心在机械臂基座系中的偏移(mm)
        self.offset_x = offset_x
        self.offset_y = offset_y
        self.offset_z = offset_z

    def camera_to_robot(self, cam_x, cam_y, cam_z):
        """相机坐标 → 机械臂基座坐标 (简化旋转变换)"""
        robot_x = cam_x + self.offset_x
        robot_y = cam_z + self.offset_y
        robot_z = -cam_y + self.offset_z
        return robot_x, robot_y, robot_z

    def quantize(self, x, y, z):
        """浮点坐标 → int8 协议范围 (-127 ~ 127)"""
        from extension.robot.protocol import TARGET_MIN, TARGET_MAX

        def clamp_to_int8(val):
            return max(TARGET_MIN, min(TARGET_MAX, int(round(val))))

        return clamp_to_int8(x), clamp_to_int8(y), clamp_to_int8(z)

    def transform(self, cam_x, cam_y, cam_z):
        """一步变换: 相机坐标 → 协议量化的机械臂坐标"""
        rx, ry, rz = self.camera_to_robot(cam_x, cam_y, cam_z)
        return self.quantize(rx, ry, rz)

    def is_workspace_valid(self, x, y, z):
        """检查是否在机械臂工作空间内"""
        dist = math.sqrt(x**2 + y**2 + z**2)
        return 50 <= dist <= 280
```

## 常见问题

**Q: 连接真机报错 "Access Denied"？**
A: 检查串口是否被其他程序（串口助手、IDE）占用。关闭后重试。

**Q: STM32 无响应？**
A: 检查固件是否正确烧录、波特率是否匹配 (115200)、TX/RX 是否交叉连接。

**Q: 坐标变换后机械臂移动到错误位置？**
A: 调整 `coordinator.py` 中 `offset_x/offset_y/offset_z` 参数，需手工标定。
