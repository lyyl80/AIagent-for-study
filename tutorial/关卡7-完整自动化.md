```python
"""多目标跟踪器: 基于 IoU 匹配 + hold 缓冲区"""
import numpy as np


class MultiTargetTracker:
    """帧间目标关联与跟踪"""

    def __init__(self, hold_frames=10, stability_threshold=5.0):
        self._tracks = {}
        self._next_id = 0
        self._hold_max = hold_frames
        self._stability_threshold = stability_threshold
        self._stable_count = {}       # track_id → 连续稳定帧数
        self._lost_count = {}         # track_id → 丢失帧数

    def update(self, detections):
        """输入当前帧检测结果, 返回关联后的跟踪目标"""
        active = {}

        current_ids = set()
        for det in detections:
            best_id = self._match(det)
            if best_id is None:
                best_id = self._next_id
                self._next_id += 1
                self._stable_count[best_id] = 0
                self._lost_count[best_id] = 0
                self._tracks[best_id] = det
            else:
                # 更新跟踪
                self._tracks[best_id] = det

                # 稳定性检测
                prev = self._tracks.get(best_id, {})
                if "X" in det and "X" in prev:
                    dx = abs(det["X"] - prev["X"])
                    dy = abs(det["Y"] - prev["Y"])
                    dz = abs(det["Z"] - prev["Z"])
                    if dx < self._stability_threshold and \
                       dy < self._stability_threshold and \
                       dz < self._stability_threshold:
                        self._stable_count[best_id] = \
                            self._stable_count.get(best_id, 0) + 1

            det["id"] = best_id
            det["stable"] = self._stable_count.get(best_id, 0)
            active[best_id] = det
            current_ids.add(best_id)

        # 丢失检测
        for tid in list(self._tracks.keys()):
            if tid not in current_ids:
                self._lost_count[tid] = self._lost_count.get(tid, 0) + 1
                if self._lost_count[tid] < self._hold_max:
                    active[tid] = self._tracks[tid]

        self._tracks = active
        return list(self._tracks.values())

    def _match(self, det):
        """在现有轨迹中查找匹配"""
        best_id = None
        best_dist = float('inf')
        for tid, track in self._tracks.items():
            dist = self._box_distance(det, track)
            if dist < best_dist and dist < 50:
                best_dist = dist
                best_id = tid
        return best_id

    def _box_distance(self, a, b):
        cx_a = a.get("cx", (a["x1"] + a["x2"]) // 2)
        cy_a = a.get("cy", (a["y1"] + a["y2"]) // 2)
        cx_b = b.get("cx", (b["x1"] + b["x2"]) // 2)
        cy_b = b.get("cy", (b["y1"] + b["y2"]) // 2)
        return np.sqrt((cx_a - cx_b) ** 2 + (cy_a - cy_b) ** 2)

    def get_stable_targets(self, min_stable_frames=5):
        """获取稳定的跟踪目标 (连续N帧未大幅移动)"""
        return [
            t for t in self._tracks.values()
            if t.get("stable", 0) >= min_stable_frames
        ]
```

```python
"""机械臂安全控制器"""
import time
from dataclasses import dataclass, field


@dataclass
class SafetyContext:
    max_retries: int = 3
    min_z: float = 200.0       # 最小安全深度 (mm)
    max_z: float = 1500.0      # 最大工作深度 (mm)
    gripper_close_ms: int = 500   # 夹爪闭合后稳定时间
    move_timeout_s: float = 10.0  # 单次移动超时
    emergency_stop: bool = False
    last_error: str = ""
    error_count: int = 0


class ArmSafetyGuard:
    """机械臂操作安全检查"""

    def __init__(self, safety_ctx=None):
        self._ctx = safety_ctx or SafetyContext()

    def check_position(self, x, y, z):
        """位置安全检查"""
        if self._ctx.emergency_stop:
            return False, "急停状态, 请先解除"
        if z < self._ctx.min_z:
            return False, f"距离过近 (Z={z:.0f} < {self._ctx.min_z})"
        if z > self._ctx.max_z:
            return False, f"距离过远 (Z={z:.0f} > {self._ctx.max_z})"
        return True, ""

    def check_gripper(self, target_angle, min_angle=75, max_angle=135):
        """夹爪角度安全检查"""
        if target_angle < min_angle or target_angle > max_angle:
            return False, f"夹爪角度越界: {target_angle} (允许 {min_angle}-{max_angle})"
        return True, ""

    def handle_error(self, error_msg):
        """错误处理"""
        self._ctx.error_count += 1
        self._ctx.last_error = error_msg

        if self._ctx.error_count >= self._ctx.max_retries:
            self._ctx.emergency_stop = True
            return "stop", f"连续{self._ctx.max_retries}次错误, 触发急停: {error_msg}"
        return "retry", f"第{self._ctx.error_count}次错误: {error_msg}"

    def reset(self):
        """重置安全状态"""
        self._ctx.emergency_stop = False
        self._ctx.error_count = 0
        self._ctx.last_error = ""
```

### 步骤 3: 任务编排器

`tools/task_orchestrator.py`：

```python
"""任务编排器: 将 AI 决策串联为有序执行序列"""
import time
from enum import Enum


class TaskState(Enum):
    PENDING = "pending"
    RUNNING = "running"
    SUCCESS = "success"
    FAILED = "failed"
    CANCELLED = "cancelled"


class GraspTask:
    """一次抓取任务的定义和执行"""

    def __init__(self, target_id, x, y, z, description=""):
        self.target_id = target_id
        self.x = x
        self.y = y
        self.z = z
        self.description = description
        self.state = TaskState.PENDING
        self.steps = []
        self.result = ""
        self.started_at = None
        self.finished_at = None

    def add_step(self, name, success):
        self.steps.append({"name": name, "success": success})

    def mark_start(self):
        self.state = TaskState.RUNNING
        self.started_at = time.time()

    def mark_success(self, msg=""):
        self.state = TaskState.SUCCESS
        self.result = msg
        self.finished_at = time.time()

    def mark_failed(self, msg=""):
        self.state = TaskState.FAILED
        self.result = msg
        self.finished_at = time.time()

    def summary(self):
        duration = ""
        if self.started_at and self.finished_at:
            duration = f" ({self.finished_at - self.started_at:.1f}s)"
        return (f"[{self.state.value}]{duration} "
                f"#{self.target_id} → {self.description}")


class TaskOrchestrator:
    """任务编排器: 管理抓取任务队列"""

    def __init__(self, max_concurrent=1):
        self._queue = []
        self._history = []
        self._max_concurrent = max_concurrent
        self._running = 0

    def submit(self, task: GraspTask):
        self._queue.append(task)

    def next(self):
        """获取下一个待执行任务"""
        for task in self._queue:
            if task.state == TaskState.PENDING:
                return task
        return None

    def execute(self, task, arm_client, coordinator, video_provider):
        """执行单个抓取任务"""
        task.mark_start()

        try:
            # Step 1: 高亮选中
            task.add_step("select", True)
            video_provider.select_bottlecap(task.target_id)

            # Step 2: 坐标变换
            rx, ry, rz = coordinator.transform(task.x, task.y, task.z)
            task.add_step("transform", True)

            # Step 3: 移动机械臂
            arm_client.send_grasp(rx, ry, rz)
            task.add_step("move", True)
            time.sleep(0.5)

            # Step 4: 确认抓取
            task.mark_success(f"瓶盖#{task.target_id} 抓取完成")
        except Exception as e:
            task.add_step("error", False)
            task.mark_failed(str(e))

        self._history.append(task)

    def get_summary(self):
        return [t.summary() for t in self._history[-10:]]
```

```python
def move_arm_to(x: float, y: float, z: float):
    client = RobotToolContext.get_client()
    coordinator = RobotToolContext.get_coordinator()

    if client is None:
        return "错误: 机械臂未连接"
    if not client.connected:
        return "错误: 串口连接已断开, 请检查硬件"

    if coordinator and not coordinator.is_workspace_valid(x, y, z):
        return (f"坐标 ({x:.0f}, {y:.0f}, {z:.0f}) "
                f"超出工作空间 (50-280mm), 请调整目标")

    try:
        if coordinator:
            rx, ry, rz = coordinator.transform(x, y, z)
        else:
            rx, ry, rz = int(x), int(y), int(z)

        client.send_grasp(rx, ry, rz)
        return (f"机械臂移动完成: 相机坐标 ({x:.0f}, {y:.0f}, {z:.0f}) "
                f"→ 机械臂坐标 ({rx}, {ry}, {rz})")
    except Exception as e:
        return f"机械臂移动失败: {e} (错误已记录, 可重试)"
```

```python
def auto_grasp_10_frames(camera, detector, tracker, arm, coordinator, orchestrator):
    """连续10帧自动检测并抓取最显著目标"""
    for i in range(10):
        color, depth, intrinsics = camera.get_frames()
        if color is None:
            continue

        raw_det = detector.detect(color, depth, intrinsics)
        for d in raw_det:
            d["color"] = detector.extract_color(color, d)

        tracked = tracker.update(raw_det)
        stable = tracker.get_stable_targets(min_stable_frames=5)

        VisionToolContext.update(tracked)

        if stable:
            target = max(stable, key=lambda d: d["confidence"])
            print(f"\n[{i}] 稳定目标 #{target['id']}: "
                  f"X={target.get('X',0):.0f} Z={target.get('Z',0):.0f}")

            task = GraspTask(
                target["id"],
                target.get("X", 0),
                target.get("Y", 0),
                target.get("Z", 0),
                f"自动抓取 #{target['id']}"
            )
            orchestrator.submit(task)

        # 执行队列中的任务
        task = orchestrator.next()
        if task:
            orchestrator.execute(task, arm, coordinator, None)
            print(orchestrator.get_summary())

    print("\n任务历史:")
    for s in orchestrator.get_summary():
        print(f"  {s}")


if __name__ == "__main__":
    camera = create_camera(use_mock=True)
    camera.start()

    detector = BottlecapDetector()
    tracker = MultiTargetTracker()
    arm = create_arm_client(use_mock=True)
    arm.connect()

    coordinator = CoordinateTransformer()
    RobotToolContext.init(arm, coordinator)

    orchestrator = TaskOrchestrator()

    auto_grasp_10_frames(camera, detector, tracker, arm, coordinator, orchestrator)

    camera.stop()
    arm.disconnect()
```

## 常见问题

**Q: 跟踪 ID 频繁跳变？**
A: 调整 `MultiTargetTracker._hold_max`（增加保持帧数）或 `_match` 的距离阈值。

**Q: 连续错误触发急停？**
A: 检查 `SafetyContext.max_retries`（默认 3），手动调用 `safety.reset()` 恢复。
